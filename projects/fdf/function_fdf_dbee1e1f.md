void print_memory_state() {
    Block *current = first_block;
    int block_count = 0;
    
    printf("\n--- État de la mémoire ---\n");
    while (current != NULL) {
        printf("Bloc %d: Adresse=%p, Taille=%zu, %s\n", 
               block_count++, 
               (void*)current, 
               current->size, 
               current->is_free ? "LIBRE" : "OCCUPÉ");
        current = current->next;
    }
    printf("------------------------\n\n");
}
