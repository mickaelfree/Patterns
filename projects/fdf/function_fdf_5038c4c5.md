t_point	project_point(t_point p, t_data *data)
{
	t_point	projected;
	
	projected = p;
	projected.x *= data->camera->zoom;
	projected.y *= data->camera->zoom;
	projected.z *= data->camera->zoom / 2;
	
	if (data->camera->projection)
		isometric(&projected.x, &projected.y, projected.z, data->camera);
	
	projected.x += data->camera->x_offset;
	projected.y += data->camera->y_offset;
	projected.color = p.color;
	
	return (projected);
}
