{

	w_ready.num = &plyr->ammo[weaponinfo[plyr->readyweapon].ammo];
    //{

    // update keycard multiple widgets
    for (i=0;i<3;i++)
    {
	keyboxes[i] = plyr->cards[i] ? i : -1;

	if (plyr->cards[i+3])
	    keyboxes[i] = i+3;

    // refresh everything if this is him coming back to life
