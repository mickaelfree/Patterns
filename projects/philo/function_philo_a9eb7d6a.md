void	free_mutex(t_data *data)
{
	if (!data)
		return ;
	if (data->use_printf)
		pthread_mutex_destroy(data->use_printf);
	free(data->use_printf);
	data->use_printf = NULL;
	if (data->write_data)
		pthread_mutex_destroy(data->write_data);
	free(data->write_data);
	data->write_data = NULL;
}
